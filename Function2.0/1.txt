    <script>
        document.getElementById('place-order-btn').addEventListener('click', () => {
            const selectedItems = [];
            const menuItems = document.querySelectorAll('.menu-item');

            menuItems.forEach(item => {
                const checkboxes = item.querySelectorAll('input[type="checkbox"]');
                const quantityInputs = item.querySelectorAll('input[type="number"]');
                const itemName = item.querySelector('h3').innerText;

                checkboxes.forEach((checkbox, index) => {
                    if (checkbox.checked) {
                        const itemPriceText = checkbox.nextSibling.nodeValue.trim(); // Get the price from the label next to the checkbox
                        const itemPrice = parseFloat(itemPriceText.replace('₱', '').split(' ')[0]); // Extract price
                        const quantity = parseInt(quantityInputs[index].value); // Get the corresponding quantity input

                        const itemTotal = itemPrice * quantity;

                        selectedItems.push({ name: `${itemName} (${itemPriceText})`, quantity: quantity, total: itemTotal });
                    }
                });
            });

            if (selectedItems.length === 0) {
                alert("Please select at least one item.");
                return; // Exit if no items are selected
            }

            const orderFormUrl = new URL('order-form.html', window.location.href);
            orderFormUrl.searchParams.append('items', JSON.stringify(selectedItems));
            orderFormUrl.searchParams.append('total', selectedItems.reduce((acc, item) => acc + item.total, 0));
            window.location.href = orderFormUrl;
        });
    </script>
